2. **Create a render wrapper function** -- add after renderer type detection, before toast:
   ```
   async function renderWithType(text) {
       if (rendererType === 'custom') {
           try {
               var currentFile = SmartBFileTree.getCurrentFile();
               await SmartBCustomRenderer.fetchAndRender(currentFile);
           } catch (e) {
               console.error('Custom renderer failed, falling back to Mermaid:', e);
               await SmartBRenderer.render(text);
           }
       } else {
           await SmartBRenderer.render(text);
       }
   }

**File 2: static/diagram-dom.js modifications**

Extend DiagramDOM to handle both Mermaid SVG and custom SVG DOM structures:

1. **Add getRendererType()** method:
   ```
   getRendererType: function() {
       var svg = this.getSVG();
       if (!svg) return null;
       // Custom renderer SVGs have a .smartb-diagram group
       if (svg.querySelector('.smartb-diagram')) return 'custom';
       return 'mermaid';
   },
   ```

2. **Update findNodeElement()** to support custom SVG:
   ```
   findNodeElement: function(nodeId) {
       var svg = this.getSVG();
       if (!svg) return null;

       // Custom renderer: direct data-node-id lookup
       var customEl = svg.querySelector('[data-node-id="' + nodeId + '"]');
       if (customEl) return customEl;

       // Mermaid renderer: pattern matching
       var elements = svg.querySelectorAll('[id]');
       for (var i = 0; i < elements.length; i++) {
           var el = elements[i];
           var id = el.getAttribute('id');
           var match = id ? id.match(NODE_RE) : null;
           if (match && match[1] === nodeId) return el;
       }
       return null;
   },
   ```

3. **Update findSubgraphElement()** to support custom SVG:
   ```
   findSubgraphElement: function(subgraphId) {
       var svg = this.getSVG();
       if (!svg) return null;

       // Custom renderer: direct data-subgraph-id lookup
       var customEl = svg.querySelector('[data-subgraph-id="' + subgraphId + '"]');
       if (customEl) return customEl;

       // Mermaid renderer: pattern matching
       var elements = svg.querySelectorAll('[id]');
       for (var i = 0; i < elements.length; i++) {
           var el = elements[i];
           var id = el.getAttribute('id');
           var match = id ? id.match(SUBGRAPH_RE) : null;
           if (match && match[1] === subgraphId) return el;
       }
       return null;
   },
   ```

4. **Update extractNodeId()** to support custom SVG:
   - At the start of the while loop, before checking id regex patterns, add:
     ```
     var dataNodeId = el.getAttribute ? el.getAttribute('data-node-id') : null;
     if (dataNodeId) return { type: 'node', id: dataNodeId };
     var dataEdgeId = el.getAttribute ? el.getAttribute('data-edge-id') : null;
     if (dataEdgeId) return { type: 'edge', id: dataEdgeId };
     var dataSubgraphId = el.getAttribute ? el.getAttribute('data-subgraph-id') : null;
     if (dataSubgraphId) return { type: 'subgraph', id: dataSubgraphId };
     ```

5. **Update getNodeLabel()** to support custom SVG:
   ```

       var el = this.findNodeElement(nodeId);
       if (!el) return null;
       // Custom renderer: text element is direct child
       var textEl = el.querySelector('text');
       if (textEl && el.hasAttribute('data-node-id')) return textEl.textContent;
       // Mermaid renderer: .nodeLabel class
       var label = el.querySelector('.nodeLabel');
       return label ? label.textContent : null;
   },
   ```

         return current;
     }
     ```

    ```typescript
    export function generateCollapsedView(
      content: string,
      subgraphs: Map<string, SubgraphInfo>,
      state: CollapseState,
      config: CollapseConfig = DEFAULT_CONFIG
    ): CollapsedDiagram {
      // Apply auto-collapse if enabled
      const autoState = autoCollapseToLimit(content, subgraphs, state, config);

      // Track which were auto-collapsed vs manual
      const autoCollapsed = [...autoState.collapsed].filter(id => !state.collapsed.has(id));
      const manualCollapsed = [...state.collapsed];

      // Generate collapsed content
      const collapsedContent = applyCollapseTransform(content, subgraphs, autoState);
      const visibleNodes = countVisibleNodes(content, subgraphs, autoState);

      return {
        content: collapsedContent,
        visibleNodes,
        autoCollapsed,
        manualCollapsed,
      };
    }
    ```

    **Update static/collapse-ui.js:**

    Add methods for auto-collapse notice:

    ```javascript
    // Add to SmartBCollapseUI object:

    autoCollapsed: [],  // IDs of auto-collapsed subgraphs

    setAutoCollapsed(ids) {
      this.autoCollapsed = ids;
      this.renderAutoCollapseNotice();
    },

    renderAutoCollapseNotice() {
      // Remove existing notice
      const existing = document.getElementById('auto-collapse-notice');
      if (existing) existing.remove();

      if (this.autoCollapsed.length === 0) return;

      const notice = document.createElement('div');
      notice.id = 'auto-collapse-notice';
      notice.className = 'auto-collapse-notice';
      notice.innerHTML = `
        <span class="notice-icon">📊</span>
        <span class="notice-text">
          ${this.autoCollapsed.length} subgraph${this.autoCollapsed.length > 1 ? 's' : ''} 
          auto-collapsed to fit ${this.config?.maxVisibleNodes || 50} node limit
        </span>
        <button class="notice-expand-all" title="Expand all">
          Expand All
        </button>
        <button class="notice-dismiss" title="Dismiss">✕</button>
      `;

      // Attach event handlers
      notice.querySelector('.notice-expand-all').addEventListener('click', () => {
        this.expandAll();
      });
      notice.querySelector('.notice-dismiss').addEventListener('click', () => {
        notice.remove();
      });

      // Insert at top of diagram container
      const container = document.getElementById('diagram-container') || document.getElementById('diagram');
      container.insertBefore(notice, container.firstChild);
    },

    expandAll() {
      // Expand all auto-collapsed subgraphs
      for (const id of this.autoCollapsed) {
        this.state.collapsed.delete(id);
      }
      this.autoCollapsed = [];
      this.renderAutoCollapseNotice();
      this.onToggle(this.getCollapsed());
    },

    setConfig(config) {
      this.config = config;
    }
    ```

    ```typescript
    // Import additional types
    import {
      parseSubgraphs,
      generateCollapsedView,
      DEFAULT_CONFIG,
      type CollapseConfig,
    } from '../diagram/collapser.js';

    // In the route handler:
    const collapsedParam = url.searchParams.get('collapsed');
    const configParam = url.searchParams.get('collapseConfig');

    let collapseConfig = { ...DEFAULT_CONFIG };
    if (configParam) {
      try {
        const userConfig = JSON.parse(configParam);
        collapseConfig = { ...DEFAULT_CONFIG, ...userConfig };
      } catch { /* use defaults */ }
    }

    const subgraphs = parseSubgraphs(diagram.mermaidContent);
    const userCollapsed = collapsedParam ? JSON.parse(collapsedParam) : [];
    const state = { collapsed: new Set<string>(userCollapsed) };

    const result = generateCollapsedView(
      diagram.mermaidContent,
      subgraphs,
      state,
      collapseConfig
    );

    // Return extended response
    return jsonResponse({
      file: diagram.file,
      mermaidContent: result.content,
      rawContent: diagram.mermaidContent,  // original content
      validation: diagram.validation,
      flags: diagram.flags,
      statuses: diagram.statuses,
      // Collapse metadata
      collapse: {
        visibleNodes: result.visibleNodes,
        autoCollapsed: result.autoCollapsed,
        manualCollapsed: result.manualCollapsed,
        config: collapseConfig,
      },
    });
    ```

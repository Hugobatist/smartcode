  <name>Task 2: Create collapse-ui.js for browser-side interaction</name>
  <files>
    static/collapse-ui.js

  <action>
    **Create static/collapse-ui.js:**

    IIFE module following the pattern of annotations.js:

    ```javascript
    (function() {
      'use strict';

      const SmartBCollapseUI = {
        state: { collapsed: new Set() },
        onToggle: null,  // callback when collapse state changes

        init(options = {}) {
          this.onToggle = options.onToggle || (() => {});
          this.attachClickHandlers();
        },

        setCollapsed(ids) {
          this.state.collapsed = new Set(ids);
        },

        getCollapsed() {
          return Array.from(this.state.collapsed);
        },

        attachClickHandlers() {
          // Delegate click events on the diagram container
          const diagram = document.getElementById('diagram');
          if (!diagram) return;

          diagram.addEventListener('click', (e) => {
            const target = e.target.closest('.node');
            if (!target) return;

            const nodeId = this.extractNodeId(target);
            if (!nodeId) return;

            // Check if this is a collapsed summary node
            if (nodeId.startsWith('__collapsed__')) {
              e.preventDefault();
              e.stopPropagation();
              const subgraphId = nodeId.replace('__collapsed__', '');
              this.expand(subgraphId);
              return;
            }

            // Check if clicking a subgraph label (for collapse)
            // Subgraph labels have class "cluster-label"
            const clusterLabel = e.target.closest('.cluster-label');
            if (clusterLabel) {
              const cluster = clusterLabel.closest('.cluster');
              if (cluster) {
                const clusterId = this.extractClusterId(cluster);
                if (clusterId) {
                  e.preventDefault();
                  e.stopPropagation();
                  this.collapse(clusterId);
                }
              }
            }
          });
        },

        extractNodeId(nodeElement) {
          // Mermaid adds id like "flowchart-NodeID-123"
          const id = nodeElement.id || '';
          const match = id.match(/flowchart-(.+?)-\d+/);
          return match ? match[1] : null;
        },

        extractClusterId(clusterElement) {
          // Clusters have id like "subgraph_ID"
          const id = clusterElement.id || '';
          return id.replace('subgraph_', '') || null;
        },

        expand(subgraphId) {
          this.state.collapsed.delete(subgraphId);
          this.onToggle(this.getCollapsed());
        },

        collapse(subgraphId) {
          this.state.collapsed.add(subgraphId);
          this.onToggle(this.getCollapsed());
        },

        toggle(subgraphId) {
          if (this.state.collapsed.has(subgraphId)) {
            this.expand(subgraphId);
          } else {
            this.collapse(subgraphId);
          }
        }
      };

      window.SmartBCollapseUI = SmartBCollapseUI;
    })();
    ```
